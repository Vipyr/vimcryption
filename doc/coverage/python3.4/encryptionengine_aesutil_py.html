def aes_matrix(): 

""" 4x4 int matrix factory 

""" 

return np.matrix(np.zeros((4, 4), dtype=int)) 

 

 

def bytes_to_matrix(byte_str): 

""" Convert a 16 character byte string to a 4x4 matrix of ints. 

""" 

matrix = aes_matrix() 

for col in range(0, 4): 

for row in range(0, 4): 

matrix[row, col] = ord(byte_str[(col * 4) + row]) 

return matrix 

 

 

def matrix_to_bytes(matrix): 

""" Convert a 4x4 matrix of ints to a 16 character byte string. 

""" 

byte_str = [] 

for column in range(0, 4): 

for row in range(0, 4): 

byte_str.append(chr(matrix[row, column])) 

return functools.reduce(lambda a, b: a + b, byte_str) 

 

 

def matrix_to_string(matrix): 

""" Convert a 4x4 matrix of ints to a hex string. 

""" 

byte_str = "" 

for column in range(0, 4): 

for row in range(0, 4): 

byte_str += "{:02x}".format(matrix[row, column]) 

return byte_str

def galois_mult(lhs, rhs): 

""" Galois multiplication function, just in case 

we want to use other bases. Not included in ALL. 

""" 

ret_value = 0 

lhs_high_bit = False 

rhs_low_bit = False 

 

for _ in range(0, 8): 

lhs_high_bit = ((lhs & 0x80) == 0x80) 

rhs_low_bit = ((rhs & 0x01) == 0x01) 

 

if rhs_low_bit: 

ret_value ^= lhs 

lhs <<= 1 

 

if lhs_high_bit: 

lhs ^= 0x1B 

rhs >>= 1 

 

return ret_value % 256
